Check if a number is an Armstrong number using Python & C++

📘 What is an Armstrong Number?

An Armstrong number is a number that is equal to the sum of its own digits each raised to the power of the number of digits.

Example: 153 = 1³ + 5³ + 3³ = 153 → Armstrong!

🧠 Python Code to Check Armstrong Number


def is_armstrong(n):
    num_str = str(n)
    power = len(num_str)
    total = sum(int(digit) ** power for digit in num_str)
    return total == n

num = 9474
print(f"{num} is Armstrong: {is_armstrong(num)}")

bool isArmstrong(int n) {
    int original = n, sum = 0, digits = 0, temp = n;
    while(temp) {
        digits++;
        temp /= 10;
    }
    temp = n;
    while(temp) {
        int digit = temp % 10;
        sum += pow(digit, digits);
        temp /= 10;
    }
    return sum == original;
}
