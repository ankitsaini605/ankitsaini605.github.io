Programming Topic #93 – C++ Classes & Objects

📦 Class & Object

class Student {
public:
  string name;
  int age;
  void show() {
    cout << name << " " << age;
  }
};
int main() {
  Student s1;
  s1.name = "Ankit";
  s1.age = 22;
  s1.show(); // Output: Ankit 22
  return 0;
}
